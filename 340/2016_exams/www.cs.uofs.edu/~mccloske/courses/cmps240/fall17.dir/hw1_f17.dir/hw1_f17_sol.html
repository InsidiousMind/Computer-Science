Problem 1 Solution:

Solution to Problem 1 
/* precondition: every element in A[0..N-1] is 
** either RED, WHITE, or BLUE
*/

r = 0;  w = 0;  b = 0;

/* loop invariant:
**   0≤r≤w≤b≤N ∧
**   every element in A[0..r-1] is RED ∧
**   every element in A[r..w-1] is WHITE ∧
**   every element in A[w..b-1] is BLUE 
*/
while (b ≠ N)  {

   if (A[b] is RED) { 
      swap(b,w); swap(w,r); 
      b=b+1; w=w+1; r=r+1;
   }
   else if (A[b] is WHITE) { 
      swap(b,w); 
      b=b+1; w=w+1;
   }
   else {   // A[b] is BLUE
      b=b+1;  
   }
}
/* post: 0≤r≤w≤b≤N ∧
**   every element in A[0..r-1] is RED ∧
**   every element in A[r..w-1] is WHITE ∧ 
**   every element in A[w..b-1] is BLUE ∧
**   b == N
*/

Solution to Problem 2 
/* precondition: every element in A[0..N-1] is 
** either RED, WHITE, or BLUE
*/

r = 0;  m = N;  w = N;

/* loop invariant:
**   0≤r≤m≤w≤N ∧ 
**   every element in A[0..r-1] is RED ∧
**   every element in A[m..w-1] is WHITE ∧ 
**   every element in A[w..N-1] is BLUE 
*/
while ( r ≠ m )  {
   if (A[r] is RED) {
      r=r+1;
   }
   else if (A[m-1] is RED) {
      swap(r,m-1); r=r+1;
   }
   else if (A[m-1] is WHITE) {
      m = m-1;
   }
   else {  // A[m-1] is BLUE
      swap(m-1,w-1); m=m-1; w=w-1;
   }
}
/* post: 0≤r≤m≤w≤N  ∧
**       every element in A[0..r-1] is RED ∧
**       every element in A[m..w-1] is WHITE ∧ 
**       every element in A[w..N-1] is BLUE ∧
**       r==m
*/
